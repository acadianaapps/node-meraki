lib/rest/static-routes

Create a Meraki REST API wrapper for the static route ressource. See the online documentation for more information.

Source:
  • rest/routes.js, line 1
Example
const apiKey = 'secret meraki api key'
const organizationId = 'meraki organization id'
const version = 'v0'
const target = 'n12'
const basePath = `/${target}/${version}/networks`
const routeEndpoints = require('./lib/rest/routes')({ apiKey, target, basePath, baseUrl, port })

Examples

Example request data

{
  "name": "VOIP",
  "subnet": "192.168.10.0/24",
  "fixedIpAssignments": {
    "13:37:de:ad:be:ef": {
      "ip": "192.168.10.5",
      "name": "fixed"
    }
  },
  "reservedIpRanges": [{
    "start": "192.168.10.20",
    "end": "192.168.10.30",
    "comment": "reserved"

}
